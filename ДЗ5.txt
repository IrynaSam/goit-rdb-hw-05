SELECT *,
(SELECT customer_id 
FROM orders 
WHERE orders.id = order_details.order_id) AS customer_id
FROM order_details


2.


SELECT *
FROM order_details
WHERE order_id IN (
    SELECT id
    FROM orders
    WHERE shipper_id = 3
);


3.


SELECT order_id, AVG(quantity) AS average_quantity
FROM (
    SELECT order_id, quantity
    FROM order_details
    WHERE quantity > 10
) AS temp_table
GROUP BY order_id;

    FROM order_details

DROP FUNCTION IF EXISTS divide_float;
DELIMITER //
CREATE FUNCTION divide_float(x FLOAT, y FLOAT) RETURNS FLOAT
DETERMINISTIC
NO SQL
BEGIN
DECLARE result FLOAT;
SET result = x / y;
RETURN result;
END //
DELIMITER ;
SELECT order_id, quantity, divide_float(quantity, 2) AS divided_quantity
FROM order_details;
